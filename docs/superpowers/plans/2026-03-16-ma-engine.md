# M&A Engine Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build an autonomous M&A intelligence engine that actively hunts healthcare deals, PE/VC activity, and acquisition targets across 7 sectors, surfacing ranked intelligence via a split-pane dashboard.

**Architecture:** Three-layer system — (1) Background signal sweep using existing MCP tools + LLM extraction, (2) Intelligence processing that resolves signals into deals and targets with vulnerability scoring, (3) On-demand deep analysis with structured dossiers and Command Center pipeline integration. Data stored in 4 new Prisma models.

**Tech Stack:** Next.js 16, React 19, Prisma + PostgreSQL, Anthropic Claude API, Framer Motion, Lucide icons, Vitest, Zod v4 (`import { z } from "zod/v4"`)

**Spec:** `docs/superpowers/specs/2026-03-16-ma-engine-design.md`

```typescript
import { prisma } from "@/lib/prisma";
import { resolveApiKey } from "@/lib/resolve-api-key";
import Anthropic from "@anthropic-ai/sdk";
import { getMCPManager } from "@/lib/mcp/client";
import { MA_SECTORS, MaSignalInputSchema, type MaSignalInput, type MaSector } from "./types";
import { SECTOR_CONFIG, getSectorQueries } from "./sectors";

// ─── Pure Functions (testable) ──────────────────────────────

/**
 * Classify signal confidence tier based on source types.
 * Tier 1: Confirmed (SEC/FTC/regulatory source)
 * Tier 2: High (2+ sources)
 * Tier 3: Signal (single source)
 */
export function classifyTier(sourceTypes: string[]): 1 | 2 | 3 {
  const hasOfficial = sourceTypes.some((t) =>
    ["SEC", "FTC", "PressRelease"].includes(t)
  );
  if (hasOfficial && sourceTypes.length >= 2) return 1;
  if (hasOfficial) return 1;
  if (sourceTypes.length >= 2) return 2;
  return 3;
}

/**
 * Deduplicate signals by headline similarity and entity overlap.
 * When duplicates exist, keeps the one with the lowest (best) tier.
 */
export function deduplicateSignals(signals: MaSignalInput[]): MaSignalInput[] {
  const seen = new Map<string, MaSignalInput>();

  for (const signal of signals) {
    const key = signal.headline.toLowerCase().trim();
    const existing = seen.get(key);
    if (!existing || signal.tier < existing.tier) {
      seen.set(key, signal);
    }
  }

  return Array.from(seen.values());
}

/**
 * Build the LLM prompt for extracting signals from search results.
 */
export function buildSweepPrompt(sector: MaSector, searchResults: string[]): string {
  const config = SECTOR_CONFIG[sector];
  return `You are an M&A signal extraction analyst. Extract healthcare M&A signals from the following search results for the "${config.name}" sector.

## Sector Context
${config.description}

## Search Results
${searchResults.map((r, i) => `[${i + 1}] ${r}`).join("\n\n")}

## Instructions
Extract each distinct M&A signal as a JSON object with these fields:
- sector: "${sector}"
- tier: 1 (confirmed deal), 2 (high confidence), or 3 (signal/rumor)
- headline: One-line summary of the signal (max 120 chars)
- summary: 2-3 sentence description with key details
- sourceUrl: URL if available, or null
- sourceType: "SEC" | "FTC" | "GDELT" | "Web" | "PressRelease"
- entitiesMentioned: Array of company/organization names mentioned

Return a JSON array of signal objects. Only include genuine M&A activity signals — deals, acquisitions, mergers, divestitures, PE transactions, regulatory actions. Ignore general industry news.

Respond with ONLY the JSON array, no other text.`;
}

// ─── Orchestrator (side-effectful) ──────────────────────────

/**
 * Run a full signal sweep across all sectors (or specified sectors).
 * Creates a MaSweep record and persists extracted signals.
 */
export async function runSweep(
  sectors: MaSector[] = [...MA_SECTORS],
): Promise<{ sweepId: string; signalsFound: number }> {
  const sweep = await prisma.maSweep.create({
    data: { sectorsSwept: sectors },
  });

  let totalSignals = 0;

  try {
    for (const sector of sectors) {
      const signals = await sweepSector(sector);
      const deduped = deduplicateSignals(signals);

      for (const signal of deduped) {
        await prisma.maSignal.create({
          data: {
            sector: signal.sector,
            tier: signal.tier,
            headline: signal.headline,
            summary: signal.summary,
            sourceUrl: signal.sourceUrl ?? null,
            sourceType: signal.sourceType,
            entitiesMentioned: signal.entitiesMentioned,
            sweepId: sweep.id,
          },
        });
      }

      totalSignals += deduped.length;
    }

    await prisma.maSweep.update({
      where: { id: sweep.id },
      data: {
        status: "completed",
        completedAt: new Date(),
        signalsFound: totalSignals,
      },
    });

    return { sweepId: sweep.id, signalsFound: totalSignals };
  } catch (error) {
    await prisma.maSweep.update({
      where: { id: sweep.id },
      data: { status: "failed", completedAt: new Date() },
    });
    throw error;
  }
}

/**
 * Sweep a single sector: run search queries via MCP, then extract signals via LLM.
 */
async function sweepSector(sector: MaSector): Promise<MaSignalInput[]> {
  const queries = getSectorQueries(sector);
  const mcpManager = getMCPManager();
  await mcpManager.initialize();

  // Collect search results from MCP tools
  const searchResults: string[] = [];

  for (const query of queries) {
    try {
      const result = await mcpManager.callTool("web_search", "search", {
        query,
        limit: 5,
      });
      if (result && typeof result === "object" && "content" in result) {
        const content = Array.isArray(result.content)
          ? result.content.map((c: { text?: string }) => c.text ?? "").join("\n")
          : String(result.content);
        searchResults.push(content);
      }
    } catch (err) {
      console.warn(`[MA Sweep] Search failed for "${query}":`, err);
    }
  }

  if (searchResults.length === 0) return [];

  // Extract signals via LLM
  const apiKey = resolveApiKey("anthropic");
  const client = new Anthropic({ apiKey });
  const prompt = buildSweepPrompt(sector, searchResults);

  const response = await client.messages.create({
    model: "claude-sonnet-4-20250514",
```

```typescript
import { VULNERABILITY_WEIGHTS, type FitVectors } from "./types";

interface SignalForScoring {
  headline: string;
  sourceType: string;
}

// Keyword patterns for each vulnerability dimension
const PATTERNS = {
  financialStress: [
    /declin/i, /loss/i, /bankrupt/i, /chapter.?11/i, /restructur/i,
    /debt/i, /credit.?downgrade/i, /revenue.?drop/i, /margin.?pressure/i,
    /impairment/i, /write.?down/i, /layoff/i, /cost.?cut/i,
  ],
  regulatoryPressure: [
    /ftc/i, /antitrust/i, /regulat/i, /compliance/i, /enforce/i,
    /investigation/i, /penalty/i, /consent.?order/i, /violation/i,
  ],
  marketPositionErosion: [
    /market.?share/i, /losing/i, /competition/i, /disrupt/i,
    /decline/i, /shrink/i, /exit/i, /divest/i,
  ],
  peInterestPatterns: [
    /private.?equity/i, /\bpe\b/i, /buyout/i, /lbo/i, /sponsor/i,
    /platform/i, /rollup/i, /roll.?up/i, /portfolio/i, /recapitaliz/i,
  ],
  managementTurnover: [
    /ceo.?resign/i, /ceo.?depart/i, /management.?change/i, /board.?shake/i,
    /executive.?exit/i, /leadership.?transition/i, /interim.?ceo/i,
  ],
};

/**
 * Compute a 0-100 vulnerability score from accumulated signals.
 * Each dimension is scored by keyword match frequency, then weighted.
 */
export function computeVulnerabilityScore(signals: SignalForScoring[]): number {
  if (signals.length === 0) return 0;

  const dimensionScores: Record<string, number> = {};

  for (const [dimension, patterns] of Object.entries(PATTERNS)) {
    let matches = 0;
    for (const signal of signals) {
      const text = signal.headline.toLowerCase();
      for (const pattern of patterns) {
        if (pattern.test(text)) {
          matches++;
          break; // Count max one match per signal per dimension
        }
      }
    }
    // Normalize: each signal can contribute up to ~10 points per dimension
    dimensionScores[dimension] = Math.min(matches * 10, 100);
  }

  const weights = VULNERABILITY_WEIGHTS;
  const weighted =
    dimensionScores.financialStress * weights.financialStress +
    dimensionScores.regulatoryPressure * weights.regulatoryPressure +
    dimensionScores.marketPositionErosion * weights.marketPositionErosion +
    dimensionScores.peInterestPatterns * weights.peInterestPatterns +
    dimensionScores.managementTurnover * weights.managementTurnover;

  // Source quality bonus: official sources boost score
  const officialCount = signals.filter((s) =>
    ["SEC", "FTC"].includes(s.sourceType)
  ).length;
  const sourceBonus = Math.min(officialCount * 5, 15);

  return Math.min(Math.round(weighted + sourceBonus), 100);
}

// Fit vector classification patterns
const FIT_PATTERNS = {
  peRollup: [/\bpe\b/i, /private.?equity/i, /rollup/i, /roll.?up/i, /platform/i, /portfolio/i, /recapitaliz/i],
  verticalIntegration: [/vertical/i, /integrat/i, /payer.*provider/i, /pbm/i, /value.?based/i],
  megaMerger: [/mega/i, /billion/i, /\$\d+b/i, /major.*merger/i, /transform/i],
  distress: [/distress/i, /bankrupt/i, /chapter.?11/i, /restructur/i, /fire.?sale/i, /insolv/i],
};

/**
 * Classify acquisition fit vectors based on signal patterns.
 * Returns 0-1 confidence for each vector type.
 */
export function classifyFitVectors(signals: SignalForScoring[]): FitVectors {
  const result: FitVectors = {
    peRollup: 0,
    verticalIntegration: 0,
    megaMerger: 0,
    distress: 0,
  };

  if (signals.length === 0) return result;

  for (const [vector, patterns] of Object.entries(FIT_PATTERNS)) {
    let matches = 0;
    for (const signal of signals) {
      for (const pattern of patterns) {
        if (pattern.test(signal.headline)) {
          matches++;
          break;
        }
      }
    }
    (result as Record<string, number>)[vector] = Math.min(matches / signals.length, 1);
  }

  return result;
}
```

```typescript
import { prisma } from "@/lib/prisma";

// ─── Pure Functions (testable) ──────────────────────────────

interface SignalRef {
  id: string;
  entitiesMentioned: string[];
}

/**
 * Group signals by normalized entity name.
 * Returns a Map of entityName → signal IDs.
 */
export function resolveEntities(signals: SignalRef[]): Map<string, string[]> {
  const groups = new Map<string, string[]>();

  for (const signal of signals) {
    for (const entity of signal.entitiesMentioned) {
      const key = entity.toLowerCase().trim();
      if (!key) continue;
      const existing = groups.get(key) ?? [];
      existing.push(signal.id);
      groups.set(key, existing);
    }
  }

  return groups;
}

/**
 * Determine if an entity should be promoted to an MaTarget.
 * Requires 2+ unlinked signals and no existing deal.
 */
export function shouldPromoteToTarget(
  signalCount: number,
  hasDeal: boolean,
): boolean {
  return signalCount >= 2 && !hasDeal;
}

/**
 * Extract headline-based trigger signal descriptions.
 */
export function extractTriggerSignals(
  signals: { headline: string }[],
): string[] {
  return signals.map((s) => s.headline);
}

// ─── Orchestrator (side-effectful) ──────────────────────────

/**
 * Process signals from a completed sweep: resolve entities, update deals,
 * create/update targets, compute vulnerability scores.
 */
export async function processSignals(sweepId: string): Promise<{
  dealsUpdated: number;
  targetsUpdated: number;
}> {
  // Fetch all unlinked signals from this sweep
  const signals = await prisma.maSignal.findMany({
    where: { sweepId, dealId: null, targetId: null },
  });

  if (signals.length === 0) return { dealsUpdated: 0, targetsUpdated: 0 };

  const entityGroups = resolveEntities(signals);
  let dealsUpdated = 0;
  let targetsUpdated = 0;

  for (const [entityName, signalIds] of entityGroups) {
    // Check for existing deal matching this entity
    const existingDeal = await prisma.maDeal.findFirst({
      where: {
        OR: [
          { acquirer: { contains: entityName, mode: "insensitive" } },
          { targetCompany: { contains: entityName, mode: "insensitive" } },
        ],
      },
    });

    if (existingDeal) {
      // Link signals to existing deal
      await prisma.maSignal.updateMany({
        where: { id: { in: signalIds } },
        data: { dealId: existingDeal.id },
      });

      // Update deal metadata
      const allDealSignals = await prisma.maSignal.findMany({
        where: { dealId: existingDeal.id },
      });
      const bestTier = Math.min(...allDealSignals.map((s) => s.tier));

      await prisma.maDeal.update({
        where: { id: existingDeal.id },
        data: {
          signalCount: allDealSignals.length,
          tier: bestTier,
          lastSignalAt: new Date(),
        },
      });

      dealsUpdated++;
      continue;
    }

    // Check for existing target
    const existingTarget = await prisma.maTarget.findFirst({
      where: {
        companyName: { contains: entityName, mode: "insensitive" },
      },
    });

    if (existingTarget) {
      // Link signals to existing target
      await prisma.maSignal.updateMany({
        where: { id: { in: signalIds } },
        data: { targetId: existingTarget.id },
      });

      // Recompute vulnerability score with all signals
      const allTargetSignals = await prisma.maSignal.findMany({
        where: { targetId: existingTarget.id },
      });
      const score = computeVulnerabilityScore(allTargetSignals);
      const fitVectors = classifyFitVectors(allTargetSignals);
      const triggers = extractTriggerSignals(allTargetSignals).slice(0, 10);

      await prisma.maTarget.update({
        where: { id: existingTarget.id },
        data: {
          vulnerabilityScore: score,
          fitVectors: fitVectors as object,
          triggerSignals: triggers,
          signalCount: allTargetSignals.length,
          lastSignalAt: new Date(),
        },
      });

      targetsUpdated++;
      continue;
    }

    // Create new target if enough signals
    const relevantSignals = signals.filter((s) => signalIds.includes(s.id));
    if (shouldPromoteToTarget(signalIds.length, false)) {
      const score = computeVulnerabilityScore(relevantSignals);
      const fitVectors = classifyFitVectors(relevantSignals);
      const triggers = extractTriggerSignals(relevantSignals);
      const sector = relevantSignals[0]?.sector ?? "hospitals";

      const target = await prisma.maTarget.create({
        data: {
          companyName: entityName,
          sector,
          vulnerabilityScore: score,
          fitVectors: fitVectors as object,
          triggerSignals: triggers,
          signalCount: signalIds.length,
          lastSignalAt: new Date(),
        },
      });

      await prisma.maSignal.updateMany({
        where: { id: { in: signalIds } },
        data: { targetId: target.id },
      });

      targetsUpdated++;
    }
  }

  // Update sweep stats
  await prisma.maSweep.update({
    where: { id: sweepId },
    data: { dealsUpdated, targetsUpdated },
  });

  return { dealsUpdated, targetsUpdated };
}
```

```typescript
function scoreColor(score: number): string {

  if (score >= 40) return "#f59e0b"; // amber

}

export default function VulnerabilityGauge({ score }: VulnerabilityGaugeProps) {
```

```typescript
import { describe, it, expect } from "vitest";
import { getEngineById, getActiveEngines } from "@/lib/engines/registry";
import { MA_SECTORS, MaSignalInputSchema, MaDealStatusSchema, FitVectorsSchema } from "@/lib/engines/ma/types";
import { SECTOR_CONFIG, getSectorQueries } from "@/lib/engines/ma/sectors";
import { classifyTier, deduplicateSignals } from "@/lib/engines/ma/sweep";
import { computeVulnerabilityScore, classifyFitVectors } from "@/lib/engines/ma/scorer";
import { resolveEntities, shouldPromoteToTarget } from "@/lib/engines/ma/processor";

describe("M&A Engine Integration", () => {
  describe("Engine registration", () => {
    it("M&A engine is active in registry", () => {
      const engine = getEngineById("ma");
      expect(engine).toBeDefined();
      expect(engine!.status).toBe("active");
      expect(engine!.route).toBe("/engines/ma");
      expect(engine!.accentColor).toBe("#8b5cf6");
    });

    it("M&A engine appears in active engines list", () => {
      const active = getActiveEngines();
      expect(active.some((e) => e.id === "ma")).toBe(true);
    });
  });

  describe("Sector coverage", () => {
    it("has configs for all 7 sectors", () => {
      expect(MA_SECTORS).toHaveLength(7);
      for (const sector of MA_SECTORS) {
        expect(SECTOR_CONFIG[sector]).toBeDefined();
        expect(getSectorQueries(sector).length).toBeGreaterThan(0);
      }
    });
  });

  describe("Signal → Deal/Target pipeline", () => {
    it("full flow: signals → classify → deduplicate → resolve → score", () => {
      // Simulate raw signals
      const signals = [
        {
          sector: "hospitals" as const,
          tier: 2,
          headline: "HCA acquires regional hospital system in Texas",
          summary: "Deal valued at $450M",
          sourceType: "Web" as const,
          entitiesMentioned: ["HCA Healthcare", "Texas Regional Health"],
        },
        {
          sector: "hospitals" as const,
          tier: 3,
          headline: "HCA acquires regional hospital system in Texas",
          summary: "Duplicate from another source",
          sourceType: "Web" as const,
          entitiesMentioned: ["HCA Healthcare"],
        },
        {
          sector: "providers" as const,
          tier: 3,
          headline: "PE firm builds provider platform with fifth acquisition",
          summary: "Rollup strategy continues",
          sourceType: "Web" as const,
          entitiesMentioned: ["Alpine Health Partners"],
        },
      ];

      // Validate all signals
      for (const s of signals) {
        expect(MaSignalInputSchema.safeParse(s).success).toBe(true);
      }

      // Classify tiers
      expect(classifyTier(["Web", "Web"])).toBe(2);
      expect(classifyTier(["SEC"])).toBe(1);

      // Deduplicate
      const deduped = deduplicateSignals(signals);
      expect(deduped).toHaveLength(2); // HCA duplicate removed

      // Entity resolution
      const refs = deduped.map((s, i) => ({
        id: `sig-${i}`,
        entitiesMentioned: s.entitiesMentioned,
      }));
      const groups = resolveEntities(refs);
      expect(groups.get("hca healthcare")).toHaveLength(1);
      expect(groups.get("alpine health partners")).toHaveLength(1);

      // Target promotion check
      expect(shouldPromoteToTarget(1, false)).toBe(false);
      expect(shouldPromoteToTarget(2, false)).toBe(true);

      // Vulnerability scoring
      const distressSignals = [
        { headline: "Hospital files chapter 11 bankruptcy", sourceType: "SEC" },
        { headline: "Credit downgrade to junk status", sourceType: "Web" },
      ];
      const score = computeVulnerabilityScore(distressSignals);
      expect(score).toBeGreaterThan(30);

      // Fit vector classification
      const fitVectors = classifyFitVectors([
        { headline: "PE firm builds provider platform via rollup", sourceType: "Web" },
      ]);
      expect(FitVectorsSchema.safeParse(fitVectors).success).toBe(true);
      expect(fitVectors.peRollup).toBeGreaterThan(0);
    });
  });

  describe("Deal status lifecycle", () => {
    it("validates all deal statuses", () => {
      const statuses = ["rumor", "announced", "under_review", "closed", "blocked", "withdrawn"];
      for (const s of statuses) {
        expect(MaDealStatusSchema.safeParse(s).success).toBe(true);
      }
    });
  });
});
```
